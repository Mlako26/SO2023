cola colaH = {}
cola colaC = {}

mutex = sem(1)
barrera = sem(0)
barreraTope = sem(0)
int heladeras = 0

heladera(i) {
    int botellas = 0
    int porrones = 0

    mutex.wait()
    colaH.encolar(i)
    heladeras++
    mutex.signal()

    while(colaH.tope() != i) {
        barrera.wait()
    }

    enchufarHeladera()
    abrirHeladera()
    cerrarHeladera()
    enfriadoRapido()
    
    colaH.desencolar()
    barrera.signal()
}
